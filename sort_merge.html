void merge(int[] array, int l, int m, int r) {

	// split the array in to a left and right sub array
	int subL =  m - l + 1; // size of the left sub array
	int subR = r - m; // size of the right sub array
	int[] arrL = new int[subL];
	int[] arrR = new int[subR];

	// populate the sub arrays
	for (int i = 0; i < subL; i++) {
		arrL[i] = array[l + i];
	}

	for (int i = 0; i < subR; i++) {
		arrR[i] = array[m + i + 1];
	}

	// subarray indices
	int i = 0;
	int j = 0;
	int k = l;

	while (i < n1 && j < n2) {
		if (arrL[i] <= arrR[j]) {
			array[k] = arrL[i];
			i++;
		} else {
			array[k] = arrR[j];
			j++;
		}

		k++;
	}

	// Remaining left subarray elements
	while ( i < subL) {
		array[k] = arrL[i];
		i++;
		k++;
	}

	// Remaining right subarray elements
	while (j < subR) {
		array[k] = subR[j];
		j++;
		k++;
	}
}
